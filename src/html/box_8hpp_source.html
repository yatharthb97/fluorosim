361  bool __attribute__((always_inline)) Invol_Check(const V &pos)
362  {
363 
364  #if FCS_INVOL_CUTOFF == 1 //Enable InVol CutOff
365  #if (FCS_VEFF_ELLIPSOID == 0) && (FCS_SYMMETRIC_BOX == 1) //Spherical and Symmetric
366  return pos.size_sq() < veff.radius_sq;
367  #elif (FCS_VEFF_ELLIPSOID == 0) && (FCS_SYMMETRIC_BOX == 0) //Spherical and Asymmetric
368  V check = pos - PSF_centre;
369  return check.size_sq() < veff.radius_sq;
370  #elif (FCS_VEFF_ELLIPSOID == 1) && (FCS_SYMMETRIC_BOX == 1) //Ellipsoid and Symmetric
371  V check = pos;
372  check.comp_divide(AD_Radius);
373  return check.size_sq() < 1.0;
374  #elif (FCS_VEFF_ELLIPSOID == 1) && (FCS_SYMMETRIC_BOX == 0) //Ellipsoid and Asymmetric
375  V check = pos - PSF_centre;
376  check.comp_divide(AD_Radius);
377  return check.size_sq() < 1.0;
378  #else
379  std::cerr << "[FATAL ERROR] Invalid State:" << __LINE__ << " : " << __FILE__ << '\n';
380  #endif
381  #elif FCS_INVOL_CUTOFF == 0 //All Particles can flash regardless of their position
382  return true; //Return true for all particles → Short Circuit
383  #endif
